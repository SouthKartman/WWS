//определение функции для ввода матрицы
double** inputm1(int n, int m)
{
    int i, j;
    double** a;
    //выделение ОП под строки
    a=new double*[n];
    if(a==NULL)
    {
        cout<<"\n Не хватает ОП";
        exit(1);
    }
    //выделение ОП под столбцы
    for(i=0; i<n; i++)
    {
        a[i]=new double[m];
        if(a[i]==NULL)
        {
            cout<<"\n Не хватает ОП";
            exit(1);
        }
    }
    //ввод элементов матрицы
    for(i=0; i<n; i++)
    {
       cout<<" Ввести "<<i+1<<"строку"<<endl;
       for(j=0; j<m; j++)
           cin>>a[i][j];
    }
    return a;
}
//функция для вывода матрицы
void outputm(double **a, int n, int m)
{
    int i, j;
    for(i=0; i<n; i++)
    {
        cout<<endl;
        for(j=0;j<m; j++)
            cout<<setw(12)<<a[i][j];
    }
}
void summa(double **a, int n, int m)
{
   int sum = 0;
   int max = 0;
    for(int i=0;i<n;i++) 
    {
        sum = 0;
        for(int j=0;j<m;j++) 
        {
            sum += a[i][j];
            if(sum>max)
        {
        max=sum;
        }
        }
        cout<<endl<< "Сумма "<<i<<" строки = "<<sum<<endl;
        
   }
cout<<" Максимальная сумма элементов = "<<max<<endl;
}
